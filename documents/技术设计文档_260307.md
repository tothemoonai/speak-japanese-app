### 1.3 技术选型理由

#### 为什么选择 Next.js 14？
- **App Router**：最新的React服务端组件架构
- **SSR/SSG**：提升首屏加载速度和SEO
- **API Routes**：前后端一体化开发
- **文件路由**：简洁的页面组织方式
- **Vercel集成**：零配置部署

#### 为什么选择 Supabase？
- **PostgreSQL**：关系型数据库，功能强大
- **实时订阅**：支持WebSocket实时更新
- **认证系统**：开箱即用的认证方案
- **Row Level Security**：数据库级别的权限控制
- **免费额度**：50,000 MAU，适合MVP

#### 为什么选择 shadcn/ui？
- **可定制性**：组件代码直接复制到项目
- **Radix UI基础**：无障碍访问支持
- **Tailwind CSS**：实用优先的样式方案
- **TypeScript**：完整的类型支持
- **现代设计**：美观且符合主流审美

#### 为什么选择 Web Speech API？
- **浏览器原生**：无需额外依赖
- **完全免费**：无调用限制
- **隐私保护**：音频无需上传到服务器
- **ASR + TTS**：同时支持语音识别和合成

**`src/services/supabase/auth.service.ts`**
```typescript
import { supabase } from '@/lib/supabase/client';
import type { User, AuthError } from '@supabase/supabase-js';

export interface RegisterInput {
  email: string;
  password: string;
  nickname?: string;
}

export interface LoginInput {
  email: string;
  password: string;
}

export interface AuthResponse {
  user: User | null;
  error: AuthError | null;
}

export class AuthService {
  /**
   * 用户注册
   * @param input 注册信息
   * @returns 用户信息和错误
   */
  async register(input: RegisterInput): Promise<AuthResponse> {
    const { email, password, nickname } = input;

    // 生成默认昵称
    const defaultNickname = nickname || email.split('@')[0];

    const { data, error } = await supabase.auth.signUp({
      email,
      password,
      options: {
        data: {
          nickname: defaultNickname,
        },
      },
    });

    return {
      user: data.user,
      error,
    };
  }

  /**
   * 用户登录
   * @param input 登录信息
   * @returns 用户信息和错误
   */
  async login(input: LoginInput): Promise<AuthResponse> {
    const { email, password } = input;

    const { data, error } = await supabase.auth.signInWithPassword({
      email,
      password,
    });

    return {
      user: data.user,
      error,
    };
  }

  /**
   * 用户登出
   */
  async logout(): Promise<{ error: AuthError | null }> {
    const { error } = await supabase.auth.signOut();
    return { error };
  }

  /**
   * 获取当前用户
   */
  async getCurrentUser(): Promise<User | null> {
    const {
      data: { user },
    } = await supabase.auth.getUser();

    return user;
  }

  /**
   * 更新用户信息
   * @param data 更新数据
   */
  async updateProfile(data: {
    nickname?: string;
    avatar_url?: string;
  }): Promise<{ error: AuthError | null }> {
    const { error } = await supabase.auth.updateUser({
      data,
    });

    return { error };
  }

  /**
   * 发送密码重置邮件
   * @param email 用户邮箱
   */
  async resetPassword(email: string): Promise<{ error: AuthError | null }> {
    const { error } = await supabase.auth.resetPasswordForEmail(email, {
      redirectTo: `${window.location.origin}/auth/reset-password`,
    });

    return { error };
  }

  /**
   * 监听认证状态变化
   * @param callback 回调函数
   */
  onAuthStateChange(
    callback: (event: string, session: any) => void
  ): () => void {
    const {
      data: { subscription },
    } = supabase.auth.onAuthStateChange((_event, session) => {
      callback(_event, session);
    });

    return () => {
      subscription.unsubscribe();
    };
  }
}

// 导出单例
export const authService = new AuthService();
```

#### 3.1.4 状态管理

**`src/store/authStore.ts`**
```typescript
import { create } from 'zustand';
import { persist } from 'zustand/middleware';
import type { User } from '@supabase/supabase-js';
import { authService } from '@/services/supabase/auth.service';

interface AuthState {
  user: User | null;
  session: any;
  isAuthenticated: boolean;
  isLoading: boolean;

  // Actions
  login: (email: string, password: string) => Promise<void>;
  register: (email: string, password: string, nickname?: string) => Promise<void>;
  logout: () => Promise<void>;
  refreshUser: () => Promise<void>;
  setUser: (user: User | null) => void;
}

export const useAuthStore = create<AuthState>()(
  persist(
    (set, get) => ({
      user: null,
      session: null,
      isAuthenticated: false,
      isLoading: false,

      login: async (email: string, password: string) => {
        set({ isLoading: true });
        try {
          const { user, error } = await authService.login({ email, password });

          if (error) throw error;

          set({
            user,
            session: null, // Session由Supabase管理
            isAuthenticated: !!user,
            isLoading: false,
          });
        } catch (error) {
          set({ isLoading: false });
          throw error;
        }
      },

      register: async (
        email: string,
        password: string,
        nickname?: string
      ) => {
        set({ isLoading: true });
        try {
          const { user, error } = await authService.register({
            email,
            password,
            nickname,
          });

          if (error) throw error;

          set({
            user,
            session: null,
            isAuthenticated: !!user,
            isLoading: false,
          });
        } catch (error) {
          set({ isLoading: false });
          throw error;
        }
      },

      logout: async () => {
        set({ isLoading: true });
        try {
          await authService.logout();
          set({
            user: null,
            session: null,
            isAuthenticated: false,
            isLoading: false,
          });
        } catch (error) {
          set({ isLoading: false });
          throw error;
        }
      },

      refreshUser: async () => {
        try {
          const user = await authService.getCurrentUser();
          set({
            user,
            isAuthenticated: !!user,
          });
        } catch (error) {
          console.error('Failed to refresh user:', error);
        }
      },

      setUser: (user: User | null) => {
        set({
          user,
          isAuthenticated: !!user,
        });
      },
    }),
    {
      name: 'auth-storage',
      partialize: (state) => ({
        user: state.user,
        isAuthenticated: state.isAuthenticated,
      }),
    }
  )
);
```

```typescript
import { useEffect } from 'react';
import { useAuthStore } from '@/store/authStore';
import { authService } from '@/services/supabase/auth.service';

export function useAuth() {
  const { user, isAuthenticated, isLoading, refreshUser, setUser } =
    useAuthStore();

  useEffect(() => {
    // 监听认证状态变化
    const unsubscribe = authService.onAuthStateChange(
      async (_event, session) => {
        if (session?.user) {
          setUser(session.user);
        } else {
          setUser(null);
        }
      }
    );

    refreshUser();

    return () => {
      unsubscribe();
    };
  }, []);

  return {
    user,
    isAuthenticated,
    isLoading,
  };
}
```

```typescript
export async function POST(request: NextRequest) {
```

#### 3.2.3 服务类

**`src/services/supabase/course.service.ts`**
```typescript
import { supabase } from '@/lib/supabase/client';
import type { Course, CourseWithProgress, CourseFilter } from '@/types/models/course';

export class CourseService {
  /**
   * 获取所有课程
   */
  async getAllCourses(): Promise<Course[]> {
    const { data, error } = await supabase
      .from('courses')
      .select('*')
      .order('sort_order', { ascending: true });

    if (error) throw error;
    return data || [];
  }

  /**
   * 根据ID获取课程
   */
  async getCourseById(id: number): Promise<Course | null> {
    const { data, error } = await supabase
      .from('courses')
      .select('*')
      .eq('id', id)
      .single();

    if (error) throw error;
    return data;
  }

  /**
   * 获取课程列表（带筛选）
   */
  async getCoursesWithFilter(
    filter: CourseFilter,
    userId?: string
  ): Promise<CourseWithProgress[]> {
    let query = supabase.from('courses').select('*');

    // 难度筛选
    if (filter.difficulty && filter.difficulty.length > 0) {
      query = query.in('difficulty', filter.difficulty);
    }

    // 主题筛选
    if (filter.theme && filter.theme.length > 0) {
      query = query.in('theme', filter.theme);
    }

    // 搜索
    if (filter.search) {
      query = query.or(`title_cn.ilike.%${filter.search}%,title_jp.ilike.%${filter.search}%`);
    }

    const { data: courses, error } = await query.order('sort_order', {
      ascending: true,
    });

    if (error) throw error;

    // 如果有用户ID，获取进度信息
    let coursesWithProgress = courses || [];
    if (userId && coursesWithProgress.length > 0) {
      const courseIds = coursesWithProgress.map((c) => c.id);

      // 获取练习记录
      const { data: practices } = await supabase
        .from('practice_records')
        .select('course_id, overall_score, completed_at')
        .eq('user_id', userId)
        .in('course_id', courseIds)
        .order('completed_at', { ascending: false });

      // 合并进度信息
      coursesWithProgress = coursesWithProgress.map((course) => {
        const coursePractices = practices?.filter(
          (p) => p.course_id === course.id
        );

        const lastPractice = coursePractices?.[0];
        const bestScore = coursePractices?.length
          ? Math.max(...coursePractices.map((p) => p.overall_score || 0))
          : 0;

        let status: 'not_started' | 'in_progress' | 'completed' =
          'not_started';
        if (coursePractices && coursePractices.length > 0) {
          status = 'completed'; // 可根据完成度调整
        }

        return {
          ...course,
          progress: lastPractice ? 100 : 0, // 简化处理
          status,
          last_practiced_at: lastPractice?.completed_at,
          best_score: bestScore,
          practice_count: coursePractices?.length || 0,
        };
      });
    }

    return coursesWithProgress;
  }

  /**
   * 获取课程主题列表
   */
  async getThemes(): Promise<string[]> {
    const { data, error } = await supabase
      .from('courses')
      .select('theme')
      .not('theme', 'is', null);

    if (error) throw error;

    const themes = [...new Set(data?.map((c) => c.theme).filter(Boolean))];
    return themes as string[];
  }

  /**
   * 获取课程详情（包含角色和句子）
   */
  async getCourseDetail(
    id: number,
    userId?: string
  ): Promise<CourseWithProgress & { characters: any[]; sentences?: any[] }> {
    const course = await this.getCourseById(id);
    if (!course) throw new Error('Course not found');

    // 获取角色
    const { data: characters } = await supabase
      .from('characters')
      .select('*')
      .eq('course_id', id)
      .order('id');

    // 获取句子（可选）
    const { data: sentences } = await supabase
      .from('sentences')
      .select('*')
      .eq('course_id', id)
      .order('sentence_order');

    // 获取用户进度
    let progress = 0;
    let status: 'not_started' | 'in_progress' | 'completed' = 'not_started';
    let bestScore = 0;
    let practiceCount = 0;

    if (userId) {
      const { data: practices } = await supabase
        .from('practice_records')
        .select('*')
        .eq('user_id', userId)
        .eq('course_id', id);

      if (practices && practices.length > 0) {
        practiceCount = practices.length;
        bestScore = Math.max(...practices.map((p) => p.overall_score || 0));
        status = 'completed';
      }
    }

    return {
      ...course,
      progress,
      status,
      best_score: bestScore,
      practice_count: practiceCount,
      characters: characters || [],
      sentences,
    };
  }
}
```

```typescript
import { useState, useRef, useCallback } from 'react';

export interface AudioRecorderOptions {
  maxDuration?: number; // 最长录制时长（秒）
  sampleRate?: number;
}

export interface Recording {
  blob: Blob;
  url: string;
  duration: number;
}

export function useAudioRecorder(options: AudioRecorderOptions = {}) {
  const { maxDuration = 30, sampleRate = 16000 } = options;

  const [isRecording, setIsRecording] = useState(false);
  const [recording, setRecording] = useState<Recording | null>(null);
  const [duration, setDuration] = useState(0);
  const [error, setError] = useState<string | null>(null);

  const mediaRecorderRef = useRef<MediaRecorder | null>(null);
  const audioChunksRef = useRef<Blob[]>([]);
  const startTimeRef = useRef<number>(0);
  const timerRef = useRef<NodeJS.Timeout | null>(null);

  /**
   * 开始录音
   */
  const startRecording = useCallback(async () => {
    try {
      // 请求麦克风权限
      const stream = await navigator.mediaDevices.getUserMedia({
        audio: {
          sampleRate,
          channelCount: 1,
          echoCancellation: true,
          noiseSuppression: true,
          autoGainControl: true,
        },
      });

      // 创建MediaRecorder
      const mediaRecorder = new MediaRecorder(stream, {
        mimeType: 'audio/webm',
      });

      mediaRecorderRef.current = mediaRecorder;
      audioChunksRef.current = [];
      setDuration(0);
      setError(null);

      // 收集音频数据
      mediaRecorder.ondataavailable = (event) => {
        if (event.data.size > 0) {
          audioChunksRef.current.push(event.data);
        }
      };

      // 录音结束处理
      mediaRecorder.onstop = () => {
        const audioBlob = new Blob(audioChunksRef.current, {
          type: 'audio/webm',
        });

        const audioUrl = URL.createObjectURL(audioBlob);

        setRecording({
          blob: audioBlob,
          url: audioUrl,
          duration: Date.now() - startTimeRef.current,
        });

        // 停止所有轨道
        stream.getTracks().forEach((track) => track.stop());
      };

      // 开始录音
      mediaRecorder.start();
      startTimeRef.current = Date.now();
      setIsRecording(true);

      // 计时器
      timerRef.current = setInterval(() => {
        const elapsed = Date.now() - startTimeRef.current;
        setDuration(Math.floor(elapsed / 1000));

        // 超时自动停止
        if (elapsed >= maxDuration * 1000) {
          stopRecording();
        }
      }, 100);
    } catch (err: any) {
      setError(err.message || 'Failed to start recording');
      console.error('Error starting recording:', err);
    }
  }, [maxDuration, sampleRate]);

  /**
   * 停止录音
   */
  const stopRecording = useCallback(() => {
    if (mediaRecorderRef.current && mediaRecorderRef.current.state !== 'inactive') {
      mediaRecorderRef.current.stop();
      setIsRecording(false);

      if (timerRef.current) {
        clearInterval(timerRef.current);
        timerRef.current = null;
      }
    }
  }, []);

  /**
   * 清除录音
   */
  const clearRecording = useCallback(() => {
    setRecording(null);
    setDuration(0);
    setError(null);
  }, []);

  /**
   * 获取音频文件
   */
  const getAudioFile = useCallback((): File | null => {
    if (!recording) return null;

    return new File([recording.blob], 'recording.webm', {
      type: 'audio/webm',
    });
  }, [recording]);

  return {
    isRecording,
    recording,
    duration,
    error,
    startRecording,
    stopRecording,
    clearRecording,
    getAudioFile,
  };
}
```

```typescript
import { useState, useRef, useCallback } from 'react';

export interface TTSOptions {
  lang?: string;
  rate?: number;
  pitch?: number;
  volume?: number;
}

export function useTTS(options: TTSOptions = {}) {
  const {
    lang = 'ja-JP',
    rate = 1.0,
    pitch = 1.0,
    volume = 1.0,
  } = options;

  const [isSpeaking, setIsSpeaking] = useState(false);
  const [isSupported, setIsSupported] = useState(true);
  const utteranceRef = useRef<SpeechSynthesisUtterance | null>(null);

  // 检查浏览器支持
  if (typeof window !== 'undefined' && !window.speechSynthesis) {
    setIsSupported(false);
  }

  /**
   * 播放语音
   */
  const speak = useCallback(
    (text: string, onStart?: () => void, onEnd?: () => void) => {
      if (!isSupported) {
        console.warn('TTS is not supported in this browser');
        return;
      }

      // 取消当前播放
      window.speechSynthesis.cancel();

      const utterance = new SpeechSynthesisUtterance(text);
      utteranceRef.current = utterance;

      utterance.lang = lang;
      utterance.rate = rate;
      utterance.pitch = pitch;
      utterance.volume = volume;

      utterance.onstart = () => {
        setIsSpeaking(true);
        onStart?.();
      };

      utterance.onend = () => {
        setIsSpeaking(false);
        onEnd?.();
      };

      utterance.onerror = (event) => {
        console.error('TTS error:', event.error);
        setIsSpeaking(false);
      };

      window.speechSynthesis.speak(utterance);
    },
    [isSupported, lang, rate, pitch, volume]
  );

  /**
   * 停止播放
   */
  const cancel = useCallback(() => {
    window.speechSynthesis.cancel();
    setIsSpeaking(false);
  }, []);

  /**
   * 暂停播放
   */
  const pause = useCallback(() => {
    if (isSpeaking) {
      window.speechSynthesis.pause();
    }
  }, [isSpeaking]);

  /**
   * 恢复播放
   */
  const resume = useCallback(() => {
    window.speechSynthesis.resume();
  }, []);

  return {
    isSpeaking,
    isSupported,
    speak,
    cancel,
    pause,
    resume,
  };
}
```
